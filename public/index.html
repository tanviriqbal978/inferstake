// Reward formula — verifiable on-chain
function _calculatePendingRewards(
    address user
) internal view returns (uint256) {

    uint256 timeElapsed =
        block.timestamp - stakes[user].stakeTimestamp;

    // reward = staked × APY × time / 365 days
    return (
        stakes[user].stakedAmount
        * apyBasisPoints
        * timeElapsed
    ) / (BASIS_POINTS_DENOMINATOR
       * SECONDS_PER_YEAR);
}

// AI advisory via 0x080C precompile
function requestAIAdvisory()
    external returns (uint256 requestId) {

    bytes memory payload = abi.encode(
        msg.sender,
        stakes[msg.sender].stakedAmount,
        apyBasisPoints
    );
    requestId = IInfernetCoordinator(
        INFERNET_COORDINATOR
    ).requestCompute(
        aiModelId, payload, 200_000
    );
}
